void Database::init(std::string host, std::string dbName, std::string user, std::string password)
{
	try {
		std::string conn_str = "host=" + host +
			" dbname=" + dbName +
			" user=" + user +
			" password=" + password;

		conn = std::make_unique<pqxx::connection>(conn_str);
	}
	catch (const std::exception& e) {
		std::cerr << "Database error: " << e.what() << "\n";
	}
	std::cout << "Connection successfull!" << std::endl;
}

void Database::insertMeasurement(int deviceID, Measurement measurement)
{
	pqxx::work txn(*conn);
	txn.exec("INSERT INTO measurements (device_id, voltage, current) "
			 "VALUES ($1, $2, $3)",
			 pqxx::params{deviceID, measurement.voltage, measurement.current});
	txn.commit();
}
void Database::cleanDatabase(int deviceID)
{
	pqxx::work txn(*conn); 
	txn.exec(
			"DELETE FROM measurements WHERE device_id = $1",
			pqxx::params{deviceID}
			);
	txn.commit();
}

void Database::printLastMeasurements(int count)
{
	pqxx::nontransaction read_txn(*conn);
	std::string read_str = 
		"SELECT m.device_id, d.name, m.recorded_at, m.voltage, m.current "
		"FROM measurements m "
		"JOIN devices d ON m.device_id = d.id "
		"ORDER BY m.recorded_at DESC LIMIT " + std::to_string(count);
	pqxx::result r = read_txn.exec(read_str);

	for (const auto& row : r) {
		int device_id = row["device_id"].as<int>();
		std::string name = row["name"].as<std::string>(); 
		std::string ts = row["recorded_at"].c_str();     
		double voltage = row["voltage"].as<double>();
		double current = row["current"].as<double>();

		std::cout
			<< device_id << " | "
			<< name << " | "
			<< ts << " | "
			<< voltage << " | "
			<< current << "\n";
	}
}
